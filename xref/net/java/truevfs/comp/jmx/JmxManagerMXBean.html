5   package net.java.truevfs.comp.jmx;
6   
7   import net.java.truevfs.kernel.spec.FsManager;
8   import net.java.truevfs.kernel.spec.FsSyncException;
9   import net.java.truevfs.kernel.spec.FsSyncWarningException;
10  
11  /**
12   * An MXBean interface for a {@linkplain FsManager file system manager}.
13   *
14   * @author Christian Schlichtherle
15   */
16  public interface JmxManagerMXBean {
17  
18      /**
19       * Returns the number of file systems which have been mounted.
20       * <p>
21       * Note that you should <em>not</em> use the returned value to synchronize
22       * conditionally - this would be unreliable!
23       *
24       * @return The number of mounted file systems.
25       */
26      int getFileSystemsMounted();
27  
28      /**
29       * Returns the total number of file systems.
30       *
31       * @return The total number of file systems.
32       */
33      int getFileSystemsTotal();
34  
35      /**
36       * Returns the number of <em>top level archive</em> file systems
37       * which have been mounted.
38       * The value of this property never exceeds
39       * {@link #getFileSystemsMounted()}.
40       * <p>
41       * Note that you should <em>not</em> use the returned value to synchronize
42       * conditionally - this would be unreliable!
43       *
44       * @return The number of mounted <em>top level archive</em> file systems.
45       */
46      int getTopLevelArchiveFileSystemsMounted();
47  
48      /**
49       * Returns the total number of <em>top level archive</em> file systems.
50       * The value of this property never exceeds
51       * {@link #getFileSystemsTotal()}.
52       *
53       * @return The total number of <em>top level archive</em> file systems.
54       */
55      int getTopLevelArchiveFileSystemsTotal();
56  
57      /**
58       * Synchronizes all file systems and eventually unmounts them.
59       * As a side effect, upon successful operation, the value of the property
60       * {@link #getFileSystemsMounted() fileSystemsMounted} is reset to zero
61       * unless some mounted file systems retain elements in their selective
62       * entry cache.
63       *
64       * @throws FsSyncWarningException if <em>only</em> warning conditions
65       *         apply.
66       *         This implies that the respective parent file system has been
67       *         synchronized with constraints, e.g. if an unclosed archive entry
68       *         stream gets forcibly closed.
69       * @throws FsSyncException if any error conditions apply.
70       */
71      void sync() throws FsSyncWarningException, FsSyncException;
72  }
