6   package net.java.truevfs.comp.tardriver;
7   
8   import java.io.BufferedOutputStream;
9   import java.io.IOException;
10  import java.io.OutputStream;
11  
12  /**
13   * Use this class to workaround issues with decorating objects which have
14   * delegates which ignore calls to {@link #close()} once it has failed before.
15   * 
16   * @author Christian Schlichtherle
17   */
18  public final class FixedBufferedOutputStream extends BufferedOutputStream {
19      private boolean ignoreClose;
20  
21      public FixedBufferedOutputStream(OutputStream out, int size) {
22          super(out, size);
23      }
24  
25      @Override
26      public void close() throws IOException {
27          if (!ignoreClose) super.close();
28      }
29  
30      public boolean getIgnoreClose() {
31          return ignoreClose;
32      }
33  
34      public void setIgnoreClose(final boolean ignoreClose) {
35          this.ignoreClose = ignoreClose;
36      }
37  }
