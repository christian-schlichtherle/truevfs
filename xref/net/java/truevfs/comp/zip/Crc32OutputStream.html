5   package net.java.truevfs.comp.zip;
6   
7   import java.io.OutputStream;
8   import java.util.zip.CRC32;
9   import java.util.zip.CheckedOutputStream;
10  import javax.annotation.CheckForNull;
11  import javax.annotation.concurrent.NotThreadSafe;
12  
13  /**
14   * An output stream which updates a CRC-32 checksum.
15   *
16   * @author Christian Schlichtherle
17   */
18  @NotThreadSafe
19  final class Crc32OutputStream extends CheckedOutputStream {
20      Crc32OutputStream(@CheckForNull OutputStream out) {
21          super(out, new CRC32());
22      }
23  }
