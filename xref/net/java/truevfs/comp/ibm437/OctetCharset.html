5   package net.java.truevfs.comp.ibm437;
6   
7   import java.nio.ByteBuffer;
8   import java.nio.CharBuffer;
9   import java.nio.charset.Charset;
10  import java.nio.charset.CharsetDecoder;
11  import java.nio.charset.CharsetEncoder;
12  import java.nio.charset.CoderResult;
13  import java.util.Arrays;
14  import javax.annotation.concurrent.Immutable;
15  
16  /**
17   * A memory efficient base class for simple 8 bit (octet) character sets.
18   *
19   * @author Christian Schlichtherle
20   */
21  @Immutable
22  public abstract class OctetCharset extends Charset {
23  
24      /**
25       * Use this character in the lookup table provided to the constructor for
26       * every character that does not have a replacement in 16 bit Unicode.
27       */
28      protected static final char REPLACEMENT = 0xFFFD;
29  
30      private final char[] byte2char;
31      private final char[][] char2byte;
32  
33      protected OctetCharset( final String cname,
34                              String[] aliases,
35                              char[] byte2char) {
36          super(cname, aliases = aliases.clone());
37          byte2char = byte2char.clone();
38  
39          // Construct sparse inverse lookup table.
40          final char[][] char2byte = new char[256][];
41          for (char i = 0; i < 256; i++) {
42              final char c = byte2char[i];
43              if (c == REPLACEMENT)
44                  continue;
45  
46              final int hi = c >>> 8;
47              final int lo = c & 0xFF;
48              char[] table = char2byte[hi];
49              if (table == null) {
50                  table = new char[256];
51                  Arrays.fill(table, REPLACEMENT);
52                  char2byte[hi] = table;
53              }
54              table[lo] = i;
55          }
56  
57          this.byte2char = byte2char;
58          this.char2byte = char2byte;
59      }
60  
61      @Override
62      public boolean contains(Charset cs) {
63          return this.getClass().isInstance(cs);
64      }
65  
66      @Override
67      public CharsetEncoder newEncoder() {
68          return new Encoder();
69      }
70  
71      private class Encoder extends CharsetEncoder {
72          Encoder() {
73              super(OctetCharset.this, 1, 1);
74          }
75  
76          @Override
77          protected CoderResult encodeLoop(   final CharBuffer in,
78                                              final ByteBuffer out) {
79              final char[][] c2b = char2byte;
80              while (in.hasRemaining()) {
81                  if (!out.hasRemaining())
82                      return CoderResult.OVERFLOW;
83                  final char c = in.get();
84                  final int hi = c >>> 8;
85                  final int lo = c & 0xFF;
86                  final char[] table = c2b[hi];
87                  final char b;
88                  if (table == null || (b = table[lo]) == REPLACEMENT) { // char is unsigned!
89                      in.position(in.position() - 1); // push back
90                      return CoderResult.unmappableForLength(1);
91                  }
92                  out.put((byte) b); // char is unsigned!
93              }
94              return CoderResult.UNDERFLOW;
95          }
96      }
97  
98      @Override
99      public CharsetDecoder newDecoder() {
100         return new Decoder();
101     }
102 
103     private class Decoder extends CharsetDecoder {
104         Decoder() {
105             super(OctetCharset.this, 1, 1);
106         }
107 
108         @Override
109         protected CoderResult decodeLoop(   final ByteBuffer in,
110                                             final CharBuffer out) {
111             final char[] b2c = byte2char;
112             while (in.hasRemaining()) {
113                 if (!out.hasRemaining())
114                     return CoderResult.OVERFLOW;
115                 final char c = b2c[in.get() & 0xFF];
116                 if (c == REPLACEMENT) {
117                     in.position(in.position() - 1); // push back
118                     return CoderResult.unmappableForLength(1);
119                 }
120                 out.put(c);
121             }
122             return CoderResult.UNDERFLOW;
123         }
124     }
125 }
