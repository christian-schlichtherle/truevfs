5   package net.java.truevfs.comp.zipdriver;
6   
7   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
8   import net.java.truecommons.cio.Entry;
9   import static net.java.truecommons.cio.Entry.Access.WRITE;
10  import static net.java.truecommons.cio.Entry.Size.DATA;
11  import static net.java.truecommons.cio.Entry.Type.DIRECTORY;
12  import static net.java.truecommons.cio.Entry.Type.FILE;
13  import net.java.truevfs.comp.zip.DateTimeConverter;
14  import net.java.truevfs.comp.zip.ZipEntry;
15  import net.java.truevfs.kernel.spec.FsArchiveEntries;
16  import net.java.truevfs.kernel.spec.FsArchiveEntry;
17  
18  /**
19   * ZIP archive entries apply the date/time conversion rules as defined by
20   * {@link DateTimeConverter#ZIP}.
21   *
22   * @see    #getDateTimeConverter()
23   * @see    ZipDriver
24   * @author Christian Schlichtherle
25   */
26  public abstract class AbstractZipDriverEntry extends ZipEntry implements FsArchiveEntry {
27  
28      protected AbstractZipDriverEntry(String name) {
29          super(name);
30          assert invariants();
31      }
32  
33      protected AbstractZipDriverEntry(String name, ZipEntry template) {
34          super(name, template);
35          assert invariants();
36      }
37  
38      private boolean invariants() {
39          assert ZipEntry.UNKNOWN == Entry.UNKNOWN;
40          return true;
41      }
42  
43      @Override
44      public Type getType() { return isDirectory() ? DIRECTORY : FILE; }
45  
46      @Override
47      protected abstract DateTimeConverter getDateTimeConverter();
48  
49      @Override
50      public long getSize(final Size type) {
51          switch (type) {
52              case DATA:
53                  return getSize();
54              case STORAGE:
55                  return getCompressedSize();
56              default:
57                  return FsArchiveEntry.UNKNOWN;
58          }
59      }
60  
61      @Override
62      public boolean setSize(final Size type, final long size) {
63          if (DATA != type) return false;
64          setSize(size);
65          return true;
66      }
67  
68      @Override
69      public long getTime(Access type) {
70          if (WRITE != type) return FsArchiveEntry.UNKNOWN;
71          long time = getTime();
72          return 0 <= time ? time : FsArchiveEntry.UNKNOWN;
73      }
74  
75      @Override
76      public boolean setTime(Access type, long time) {
77          if (WRITE != type) return false;
78          setTime(time);
79          return true;
80      }
81  
82      @Override
83      @SuppressFBWarnings("NP_BOOLEAN_RETURN_NULL")
84      public Boolean isPermitted(Access type, Entity entity) { return null; }
85  
86      @Override
87      public boolean setPermitted(Access type, Entity entity, Boolean value) {
88          return false;
89      }
90  
91      /**
92       * Returns a string representation of this object for debugging and logging
93       * purposes.
94       */
95      @Override
96      public String toString() { return FsArchiveEntries.toString(this); }
97  }
