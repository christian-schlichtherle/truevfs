5   package net.java.truevfs.comp.zipdriver;
6   
7   import java.nio.charset.Charset;
8   import javax.annotation.concurrent.Immutable;
9   import net.java.truevfs.comp.zip.DateTimeConverter;
10  import net.java.truevfs.comp.zip.ZipEntry;
11  
12  /**
13   * An archive driver for ZIP files.
14   * By default, ZIP files use the IBM437 character set for the encoding of entry
15   * names and comments (unless the General Purpose Bit 11 is present in
16   * accordance with appendix D of the
17   * <a href="http://www.pkware.com/documents/casestudies/APPNOTE.TXT">ZIP File Format Specification</a>).
18   * They also apply the date/time conversion rules according to
19   * {@link DateTimeConverter#ZIP}.
20   * This configuration pretty much constrains the applicability of this driver
21   * to North American and Western European countries.
22   * However, this driver generally provides best interoperability with third
23   * party tools like the Windows Explorer, WinZip, 7-Zip etc.
24   * To some extent this applies even outside these countries.
25   * Therefore, while you should use this driver to access plain old ZIP files,
26   * you should <em>not</em> use it for custom application file formats - use the
27   * {@link JarDriver} instead in this case.
28   * <p>
29   * This driver does <em>not</em> check the CRC value of any entries in existing
30   * archives - use {@link CheckedZipDriver} instead.
31   * <p>
32   * Sub-classes must be thread-safe and should be immutable!
33   *
34   * @author Christian Schlichtherle
35   */
36  @Immutable
37  public class ZipDriver extends AbstractZipDriver<ZipDriverEntry> {
38  
39      /**
40       * The character set for entry names and comments in &quot;traditional&quot;
41       * ZIP files, which is {@code "IBM437"}.
42       */
43      public static final Charset ZIP_CHARSET = Charset.forName("IBM437");
44  
45      /**
46       * {@inheritDoc}
47       * 
48       * @return {@link #ZIP_CHARSET}.
49       */
50      @Override
51      public Charset getCharset() {
52          return ZIP_CHARSET;
53      }
54  
55      @Override
56      public ZipDriverEntry newEntry(String name) {
57          return new ZipDriverEntry(name);
58      }
59  
60      @Override
61      public ZipDriverEntry newEntry(String name, ZipEntry template) {
62          return new ZipDriverEntry(name, template);
63      }
64  }
