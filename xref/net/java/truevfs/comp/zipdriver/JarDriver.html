5   package net.java.truevfs.comp.zipdriver;
6   
7   import java.nio.charset.Charset;
8   import javax.annotation.concurrent.Immutable;
9   import net.java.truevfs.comp.zip.DateTimeConverter;
10  import net.java.truevfs.comp.zip.ZipEntry;
11  
12  /**
13   * An archive driver for Java Archive files (JAR).
14   * JAR files use the UTF-8 character set for the encoding of entry
15   * names and comments.
16   * They also apply the date/time conversion rules according to
17   * {@link DateTimeConverter#JAR}.
18   * This configuration makes this driver applicable for all countries.
19   * However, it pretty much constrains the interoperability of this driver to
20   * Java applications and Info-ZIP.
21   * Therefore, while you should <em>not</em> use this driver to access plain old
22   * ZIP files, you should definitely use it for custom application file formats
23   * <p>
24   * Other than this, JAR files are treated like plain old ZIP files.
25   * In particular, this class does <em>not</em> check or ensure a certain
26   * directory structure or the existance of certain entries (e.g.
27   * {@code META-INF/MANIFEST.MF}) within a JAR file.
28   * <p>
29   * This driver does <em>not</em> check the CRC value of any entries in existing
30   * archives - use {@link CheckedJarDriver} instead.
31   * <p>
32   * Sub-classes must be thread-safe and should be immutable!
33   * 
34   * @author Christian Schlichtherle
35   */
36  @Immutable
37  public class JarDriver extends AbstractZipDriver<JarDriverEntry> {
38  
39      /**
40       * The character set for entry names and comments in JAR files, which is
41       * {@code "UTF-8"}.
42       */
43      public static final Charset JAR_CHARSET = Charset.forName("UTF-8");
44  
45      /**
46       * {@inheritDoc}
47       * 
48       * @return {@link #JAR_CHARSET}.
49       */
50      @Override
51      public Charset getCharset() {
52          return JAR_CHARSET;
53      }
54  
55      @Override
56      public JarDriverEntry newEntry(String name) {
57          return new JarDriverEntry(name);
58      }
59  
60      @Override
61      public JarDriverEntry newEntry(String name, ZipEntry template) {
62          return new JarDriverEntry(name, template);
63      }
64  }
