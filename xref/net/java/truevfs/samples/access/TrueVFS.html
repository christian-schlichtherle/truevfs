35  /**
36   * A comprehensive command line utility which allows you to work
37   * with entries in all supported archive files using Unix like commands
38   * ({@code cat}, {@code cp}, {@code rm}, {@code mkdir},
39   * {@code rmdir}, {@code ls} etc.).
40   * <p>
41   * Please note that this utility class features some optional archive drivers
42   * which provide additional safety or otherwise unavailable features.
43   * Some of these features are not normally used because of their negative
44   * performance impact and hence this utility class should not be used for a
45   * performance benchmark.
46   * <p>
47   * For example, the ZIP drivers used in this utility <em>always</em> check
48   * the CRC-32 values provided in the ZIP file.
49   * In addition, the SFX driver is used which allows you to browse
50   * {@code .exe} files if they happen to be SelF eXtracting archives (SFX).
51   * If they are not however, some considerable amount of time is spent searching
52   * for the Central Directory required to be present in ZIP (and hence SFX)
53   * files.
54   *
55   * @author Christian Schlichtherle
56   */

252     /**
253      * Scans the given command parameters for options of the given class.
254      * As a side effect, any found options are popped off the parameter stack.
255      *
256      * @param  <T> the type of the enum class for the options.
257      * @param  args the command arguments.
258      * @param  option the enum class for the options.
259      * @return a bit field of the options found.
260      */
261     private static <T extends Enum<T>> BitField<T> options(
262             final Deque<String> args,
263             final Class<T> option) {
264         BitField<T> options = BitField.noneOf(option);
265         for (   String arg;
266                 null != (arg = args.peek()) && '-' == arg.charAt(0);
267                 args.pop()) {
268             arg = arg.substring(1).toUpperCase(Locale.ENGLISH);
269             options = options.set(valueOf(option, arg));
270         }
271         return options;
272     }

274     void cpOrMv(
275             final BitField<CpOption> options,
276             final Deque<String> args)
277             throws IOException {
278         try (final TConfig config = TConfig.open()) {
279             TArchiveDetector srcDetector;
280             if (options.get(CpOption.CP437IN))
281                 srcDetector = newArchiveDetector(Charset.forName("IBM437"));
282             else if (options.get(CpOption.UTF8IN))
283                 srcDetector = newArchiveDetector(Charsets.UTF_8);
284             else
285                 srcDetector = config.getArchiveDetector();
286 
287             TArchiveDetector dstDetector;
288             if (options.get(CpOption.UNZIP))
289                 dstDetector = TArchiveDetector.NULL;
290             else if (options.get(CpOption.CP437OUT))
291                 dstDetector = newArchiveDetector(Charset.forName("IBM437"));
292             else if (options.get(CpOption.UTF8OUT))
293                 dstDetector = newArchiveDetector(Charsets.UTF_8);
294             else
295                 dstDetector = config.getArchiveDetector();
296 
297             config.setAccessPreferences(config.getAccessPreferences()
298                     .set(FsAccessOption.STORE, options.get(CpOption.STORE))
299                     .set(FsAccessOption.COMPRESS, options.get(CpOption.COMPRESS))
300                     .set(FsAccessOption.GROW, options.get(CpOption.GROW))
301                     .set(FsAccessOption.ENCRYPT, options.get(CpOption.ENCRYPT)));
302 
303             final TFile last = new TFile(args.removeLast(), dstDetector);
304             final boolean expandPath = last.isDirectory();
305             if (args.isEmpty() || 1 < args.size() && !expandPath)
306                 throw new IllegalArgumentException();
307 
308             for (final String arg : args) {
309                 final TFile src = new TFile(arg, srcDetector);
310                 final TFile dst = expandPath
311                         ? new TFile(last, src.getName(), dstDetector)
312                         : last;
313                 if (equals(MV)) {
314                     try {
315                         if (dst.isFile()) dst.rm();
316                         src.mv(dst);
317                     } catch (final IOException ex) {
318                         throw new IOException(message("cmt", src, dst), ex);
319                     }
320                 } else { // cp
321                     if (options.get(CpOption.R))
322                         if (options.get(CpOption.P))
323                             TFile.cp_rp(src, dst, srcDetector, dstDetector);
324                         else
325                             TFile.cp_r(src, dst, srcDetector, dstDetector);
326                     else
327                     if (options.get(CpOption.P))
328                         TFile.cp_p(src, dst);
329                     else
330                         TFile.cp(src, dst);
331                 }
332             }
333         }
334     }

397     void ls(
398             final BitField<LsOption> options,
399             final Deque<String> args)
400             throws IOException {
401         if (0 >= args.size()) args.push(".");
402         final boolean multi = 1 < args.size();
403         for (final String arg : args) {
404             final TFile file = new TFile(arg);
405             if (multi) out.println(arg + ":");
406             if (file.isDirectory()) ls(options, file, "");
407             else ls(options, file, file.getPath());
408         }
409     }

411     private void ls(
412             final BitField<LsOption> options,
413             final TFile file,
414             final String path)
415             throws IOException {
416         if (file.isDirectory()) {
417             final TFile[] members = file.listFiles();
418             if (null == members)
419                 throw new IOException(message("dina", path));
420             // Sort directories to the start.
421             Arrays.sort(members, new TFileComparator());
422             for (final TFile member : members) {
423                 String memberPath = member.getName();
424                 if (!path.isEmpty())
425                     memberPath = path + TFile.separator + memberPath;
426                 ls(options, member.toPath(), memberPath);
427                 if (options.get(LsOption.R) && member.isDirectory())
428                     ls(options, member, memberPath);
429             }
430         } else if (file.exists()) {
431             ls(options, file.toPath(), path);
432         } else {
433             throw new IOException(message("nsfod", path));
434         }
435     }

437     private static void ls(
438             final BitField<LsOption> options,
439             final TPath file,
440             final String path)
441             throws IOException {
442         final BasicFileAttributes attr =
443                 readAttributes(file, BasicFileAttributes.class);
444         final boolean detailed = options.get(LsOption.L);
445         if (detailed)
446             out.printf("%,11d %tF %<tT ", attr.size(), attr.lastModifiedTime().toMillis());
447         out.append(path);
448         if (detailed) {
449             if (attr.isDirectory()) {
450                 if (attr.isRegularFile()) out.append('+'); // covariant
451                 else out.append(TFile.separator);
452             }
453             if (attr.isSymbolicLink()) out.append('>');
454             if (attr.isOther()) out.append('?');
455         }
456         out.println();
457     }

463     @SuppressWarnings("CallToThreadDumpStack")
464     private static int main(final Deque<String> args) {
465         final TrueVFS command;
466         try {
467             command = valueOf(args.pop().toUpperCase(Locale.ENGLISH));
468         } catch (final IllegalArgumentException | NoSuchElementException ex) {
469             final StringBuilder builder = new StringBuilder(25 * 80);
470             for (final TrueVFS truevfs : values()) {
471                 if (0 != builder.length()) builder.append('\n');
472                 builder.append(truevfs.getSynopsis());
473             }
474             err.println(builder.toString());
475             return 1;
476         }
477         try {
478             command.run(args);
479         } catch (final IllegalArgumentException | NoSuchElementException ex) {
480             err.println(command.getUsage());
481             return 2;
482         } catch (IOException ex) {
483             ex.printStackTrace();
484             return 3;
485         }
486         return 0;
487     }

514     private static class RmVisitor extends SimpleFileVisitor<Path> {
515         @Override
516         public FileVisitResult visitFile(
517                 final Path file,
518                 final BasicFileAttributes attrs)
519         throws IOException {
520             delete(file);
521             return FileVisitResult.CONTINUE;
522         }
523 
524         @Override
525         public FileVisitResult postVisitDirectory(
526                 final Path dir,
527                 final @Nullable IOException exc)
528         throws IOException {
529             if (null != exc) throw exc;
530             delete(dir);
531             return FileVisitResult.CONTINUE;
532         }
