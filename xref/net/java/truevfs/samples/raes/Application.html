5   package net.java.truevfs.samples.raes;
6   
7   import java.io.IOException;
8   import static java.lang.System.*;
9   import java.util.NoSuchElementException;
10  import java.util.ResourceBundle;
11  
12  /**
13   * @author Christian Schlichtherle
14   */
15  public abstract class Application {
16  
17      String message(String key, Object... args) {
18          return String.format(
19                  ResourceBundle.getBundle(getClass().getName()).getString(key),
20                  args);
21      }
22  
23      int run(final String[] args) throws IOException {
24          try {
25              runChecked(args);
26          } catch (IllegalArgumentException | NoSuchElementException ex) {
27              String message = ex.getLocalizedMessage();
28              if (null == message) message = message("usage", getClass().getSimpleName());
29              err.println(message);
30              return 2;
31          } catch (Exception ex) {
32              ex.printStackTrace();
33              return 1;
34          }
35          return 0;
36      }
37  
38      abstract void runChecked(String[] args) throws IOException;
39  }
