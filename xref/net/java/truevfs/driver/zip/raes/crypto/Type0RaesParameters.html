5   package net.java.truevfs.driver.zip.raes.crypto;
6   
7   import net.java.truecommons.key.spec.common.AesKeyStrength;
8   
9   /**
10   * The parameters of this interface are used with RAES <i>type 0</i> files.
11   * Type 0 RAES files use password based encryption according to the
12   * specifications in PKCS #5 V2.0 und PKCS #12 V1.0.
13   * <p>
14   * Implementations do not need to be safe for multi-threading.
15   *
16   * @see     <a href="http://www.rsasecurity.com/rsalabs/pkcs/pkcs-5/index.html">PKCS #5</a>
17   * @see     <a href="http://www.rsasecurity.com/rsalabs/pkcs/pkcs-12/index.html">PKCS #12</a>
18   * @author  Christian Schlichtherle
19   */
20  public interface Type0RaesParameters extends RaesParameters {
21  
22      /**
23       * Returns the password to use for writing a RAES type 0 file.
24       *
25       * @return A clone of the char array holding the password to use
26       *         for writing a RAES type 0 file.
27       * @throws RaesKeyException If key retrieval has failed for some reason.
28       */
29      char[] getPasswordForWriting() throws RaesKeyException;
30  
31      /**
32       * Returns the password to use for reading a RAES type 0 file.
33       * This method is called consecutively until either the returned password
34       * is successfully validated or an exception is thrown.
35       *
36       * @param  invalid {@code true} iff a previous call to this method returned
37       *         an invalid password.
38       * @return A clone of the char array holding the password to use
39       *         for reading a RAES type 0 file.
40       * @throws RaesKeyException If key retrieval has failed for some reason.
41       */
42      char[] getPasswordForReading(boolean invalid) throws RaesKeyException;
43  
44      /**
45       * Returns the key strength to use for writing a RAES type 0 file.
46       *
47       * @return The key strength to use for writing a RAES type 0 file.
48       * @throws RaesKeyException If key retrieval has failed for some reason.
49       */
50      AesKeyStrength getKeyStrength() throws RaesKeyException;
51  
52      /**
53       * Sets the key strength obtained from reading a RAES type 0 file.
54       *
55       * @param  keyStrength the key strength obtained from reading a RAES type 0
56       *         file.
57       * @throws RaesKeyException If key retrieval has failed for some reason.
58       */
59      void setKeyStrength(AesKeyStrength keyStrength) throws RaesKeyException;
60  }
