5   package net.java.truevfs.kernel.spec;
6   
7   import java.util.Comparator;
8   import javax.annotation.concurrent.NotThreadSafe;
9   import net.java.truecommons.shed.PriorityExceptionBuilder;
10  
11  /**
12   * Assembles an {@link FsSyncException} from one or more sync exceptions by
13   * {@linkplain Exception#addSuppressed(Throwable) suppressing} and optionally
14   * {@linkplain FsSyncException#getPriority() prioritizing} them.
15   *
16   * @author Christian Schlichtherle
17   */
18  @NotThreadSafe
19  public final class FsSyncExceptionBuilder
20  extends PriorityExceptionBuilder<FsSyncException> {
21  
22      public FsSyncExceptionBuilder() {
23          super(FsSyncExceptionComparator.INSTANCE);
24      }
25  
26      private static final class FsSyncExceptionComparator
27      implements Comparator<FsSyncException> {
28          static final FsSyncExceptionComparator
29                  INSTANCE = new FsSyncExceptionComparator();
30  
31          @Override
32          public int compare(FsSyncException o1, FsSyncException o2) {
33              final int p1 = o1.getPriority();
34              final int p2 = o2.getPriority();
35              return p1 < p2 ? -1 : p1 == p2 ? 0 : 1;
36          }
37      }
38  }
