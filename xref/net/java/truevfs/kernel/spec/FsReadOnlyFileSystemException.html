5   package net.java.truevfs.kernel.spec;
6   
7   import java.nio.file.FileSystemException;
8   import javax.annotation.concurrent.ThreadSafe;
9   
10  /**
11   * Indicates that an I/O operation was trying to modify a read-only file system.
12   * <p>
13   * The primary difference between this exception class and
14   * {@link java.nio.file.ReadOnlyFileSystemException} is that this class is a
15   * subclass of {@code FileSystemException} while the latter is a subclass of
16   * {@link java.lang.UnsupportedOperationException}.
17   */
18  @ThreadSafe
19  public class FsReadOnlyFileSystemException extends FileSystemException {
20      private static final long serialVersionUID = 987645923519873262L;
21  
22      public FsReadOnlyFileSystemException(FsMountPoint mountPoint) {
23          super(mountPoint.toString());
24      }
25  
26      public FsReadOnlyFileSystemException(FsMountPoint mountPoint, Throwable cause) {
27          super(mountPoint.toString());
28          super.initCause(cause);
29      }
30  }
