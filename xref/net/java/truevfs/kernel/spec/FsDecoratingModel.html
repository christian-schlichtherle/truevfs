5   package net.java.truevfs.kernel.spec;
6   
7   import javax.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * An abstract decorator for a file system model.
11   *
12   * @author Christian Schlichtherle
13   */
14  @ThreadSafe
15  public abstract class FsDecoratingModel extends FsAbstractModel {
16  
17      /** The decorated file system model. */
18      protected final FsModel model;
19  
20      protected FsDecoratingModel(final FsModel model) {
21          super(model.getMountPoint(), model.getParent());
22          this.model = model;
23      }
24  
25      @Override
26      public boolean isMounted() { return model.isMounted(); }
27  
28      @Override
29      public void setMounted(boolean touched) { model.setMounted(touched); }
30  
31      /**
32       * Returns a string representation of this object for debugging and logging
33       * purposes.
34       */
35      @Override
36      public String toString() {
37          return String.format("%s@%x[model=%s]",
38                  getClass().getName(),
39                  hashCode(),
40                  model);
41      }
42  }
