5   package net.java.truevfs.access;
6   
7   import edu.umd.cs.findbugs.annotations.CleanupObligation;
8   import edu.umd.cs.findbugs.annotations.CreatesObligation;
9   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
10  
11  import java.io.File;
12  import java.io.FileWriter;
13  import java.io.IOException;
14  import java.io.OutputStreamWriter;
15  import java.nio.charset.Charset;
16  import java.nio.charset.CharsetEncoder;
17  import javax.annotation.concurrent.Immutable;
18  
19  /**
20   * A replacement for the class {@link FileWriter} for writing plain old files
21   * or entries in an archive file.
22   * Mind that applications cannot write archive files directly - just their
23   * entries!
24   *
25   * @see    TConfig#setLenient
26   * @see    TFileReader
27   * @author Christian Schlichtherle
28   */
29  @Immutable
30  @CleanupObligation
31  public final class TFileWriter extends OutputStreamWriter {
32  
33      /**
34       * Constructs a new {@code TFile} writer.
35       * This writer uses the default character set for encoding characters
36       * to bytes.
37       * 
38       * @param  file a file to write.
39       * @throws IOException on any I/O error.
40       */
41      @CreatesObligation
42      @SuppressFBWarnings("DM_DEFAULT_ENCODING" )
43      public TFileWriter(File file) throws IOException {
44          super(new TFileOutputStream(file));
45      }
46  
47      /**
48       * Constructs a new {@code TFile} writer.
49       * This writer uses the default character set for encoding characters
50       * to bytes.
51       * 
52       * @param  file a file to write.
53       * @param  append iff {@code true}, then this writer appends the data to the
54       *         given file.
55       * @throws IOException on any I/O error.
56       */
57      @CreatesObligation
58      @SuppressFBWarnings("DM_DEFAULT_ENCODING")
59      public TFileWriter(File file, boolean append) throws IOException {
60          super(new TFileOutputStream(file, append));
61      }
62  
63      /**
64       * Constructs a new {@code TFile} writer.
65       * 
66       * @param  file a file to write.
67       * @param  append iff {@code true}, then this writer appends the data to the
68       *         given file.
69       * @param  charset a character set for encoding characters to bytes.
70       * @throws IOException on any I/O error.
71       */
72      @CreatesObligation
73      public TFileWriter(File file, boolean append, Charset charset)
74      throws IOException {
75          super(new TFileOutputStream(file, append), charset);
76      }
77  
78      /**
79       * Constructs a new {@code TFile} writer.
80       * 
81       * @param  file a file to write.
82       * @param  append iff {@code true}, then this writer appends the data to the
83       *         given file.
84       * @param  encoder an encoder for encoding characters to bytes.
85       * @throws IOException on any I/O error.
86       */
87      @CreatesObligation
88      public TFileWriter(File file, boolean append, CharsetEncoder encoder)
89      throws IOException {
90          super(new TFileOutputStream(file, append), encoder);
91      }
92  }
